

oracle11g手动建库

1. 设置环境变量
export ORACLE_BASE=/u01/app/Oracle
export ORACLE_HOME=/u01/app/oracle/product/11.2.0.3/db_1
export ORACLE_SID=orcl
export PATH=$ORACLE_HOME/bin:$ORACLE_HOME/OPatch:$PATH
export PATH

2. 创建orcl目录
[oracle@ogg04 ~]$ pwd

6.创建spfile
SQL> create spfile  from pfile='$ORACLE_HOME/dbs/initorcl.ora';
File created.

7.建库脚本
su - root
[root@oracle ~]# mkdir -p /oradata/orcl/
[root@oracle ~]# chown -R oracle.oinstall /oradata/orcl

SQL> create database orcl CONTROLFILE REUSE 
   MAXINSTANCES 8 
   MAXDATAFILES 2000
   MAXLOGHISTORY 1024
   MAXLOGFILES 64
   MAXLOGMEMBERS 4
   character set ZHS16GBK national character set AL16UTF16
   logfile 
   group 1 ('/oradata/orcl/redo01.log') size 50M reuse,
   group 2 ('/oradata/orcl/redo02.log') size 50M reuse,
   group 3 ('/oradata/orcl/redo03.log') size 50M reuse
datafile '/oradata/orcl/system01.dbf' size 1000M reuse extent management local
sysaux datafile '/oradata/orcl/sysaux01.dbf' size 800M
default temporary tablespace temp tempfile '/oradata/orcl/temp01.dbf' size 500M reuse
undo tablespace undotbs1 datafile '/oradata/orcl/undotbs01.dbf' size 126M reuse
USER SYS IDENTIFIED BY "oracle" USER SYSTEM IDENTIFIED BY "oracle";
 
Database created.

8. 运行数据字典脚本，其中catalog和catproc是必需的，其它可选：
SQL> spool /home/oracle/cat_orcl.log
SQL> @?/rdbms/admin/catalog.sql ;      （建数据字典视图）
SQL> @?/rdbms/admin/catproc.sql ;      （建存储过程包）
SQL> @?/rdbms/admin/catblock.sql ;     （建锁相关的几个视图）
SQL> @?/rdbms/admin/catoctk.sql ;      （建密码工具包dbms_crypto_toolkit）
SQL> @?/rdbms/admin/owminst.plb ;      （建工作空间管理相关对象，如dmbs_wm）
SQL> spool off
执行完后检查/orahome/cat.log看看有什么不可接受的错误没有。
 
9. 新建sqlplus属性和帮助、USERS表空间
SQL> connect system/oracle
SQL> @?/sqlplus/admin/pupbld.sql
SQL> @?/sqlplus/admin/help/hlpbld.sql helpus.sql
SQL> connect /as sysdba
SQL> CREATE TABLESPACE USERS LOGGING DATAFILE '/oradata/orcl/user01.dbf' SIZE 100M EXTENT MANAGEMENT LOCAL SEGMENT SPACE MANAGEMENT AUTO;
SQL> ALTER DATABASE DEFAULT TABLESPACE USERS;
